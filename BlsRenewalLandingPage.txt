import React, { useEffect, useState } from "react";

const BOOKING_IFRAME_SRC =
  "https://cprwestcovina-commits.github.io/bls-booking/bls-renewal.html";

function RenewalBookingIframe({ minHeight = 380 }) {
  const [height, setHeight] = useState(minHeight);

  useEffect(() => {
    function onMessage(e) {
      if (e?.data?.type === "renewal-widget-resize" && typeof e.data.height === "number") {
        setHeight(Math.max(minHeight, e.data.height));
      }
    }
    window.addEventListener("message", onMessage);
    return () => window.removeEventListener("message", onMessage);
  }, [minHeight]);

  return (
    <iframe
      src={BOOKING_IFRAME_SRC}
      title="BLS Renewal Booking"
      loading="lazy"
      style={{
        width: "100%",
        maxWidth: "560px",
        height: `${height}px`,
        border: "none",
        display: "block",
        transition: "height 0.2s ease",
      }}
      className="mx-auto"
    />
  );
}
